Rust client for Koios API

Initial version created using [openapi-generator-cli](https://github.com/OpenAPITools/openapi-generator):

    npm install @openapitools/openapi-generator-cli -g
    wget https://api.koios.rest/koiosapi.yaml
    npx @openapitools/openapi-generator-cli generate -i koiosapi.yaml -g rust -o koios_api_client

example use:
```rust
let mut cfg = koios_api_client::apis::configuration::Configuration::new();
cfg.base_path = "https://preprod.koios.rest/api/v0".into();

let tips = koios_api_client::apis::network_api::tip_get(&cfg).await.unwrap();
let tip = tips.get(0).unwrap();

let params = koios_api_client::apis::epoch_api::epoch_params_get(&cfg, Some(&tip.epoch_no.unwrap().to_string())).await.unwrap();

let params_for_this_epoch = params.get(0).unwrap();

println!("Result: {:?}",params_for_this_epoch);
```
